# Observation Table

CREATE TABLE observation (
   id INT NOT NULL AUTO_INCREMENT,
   rm DECIMAL (3,1),
   obstype ENUM('discharge', 'dryness'),
   datet DATETIME,
   PRIMARY KEY (id),
   INDEX par_ind (rm),
   FOREIGN KEY (rm)
     REFERENCES rivermile(rm)
     ON UPDATE CASCADE
     ON DELETE CASCADE
) ENGINE=INNODB;

# Dryness Table
CREATE TABLE dryness (
   id INT,
   rmdown DECIMAL (3,1),
   PRIMARY KEY (id),
   FOREIGN KEY (id)
     REFERENCES observation(id)
     ON UPDATE CASCADE
     ON DELETE CASCADE
) ENGINE=INNODB;

# Q Table
CREATE TABLE discharge (
   id INT,
   qtype ENUM('usgs', 'estimate'),
   q DECIMAL (4,2),
   PRIMARY KEY (id),
   FOREIGN KEY (id)
     REFERENCES observation(id)
     ON UPDATE CASCADE
     ON DELETE CASCADE
) ENGINE=INNODB;

# return the dates of observations at specific river miles
SELECT 
  a.rm, b.datet
FROM rivermile a
INNER JOIN 
  observation b
ON 
  a.rm = b.rm
ORDER BY
  a.rm
;

# count the number and types of dry samples per return
SELECT 
  a.rm, b.obstype, COUNT(b.datet) as countd
FROM rivermile a
LEFT JOIN 
  observation b
ON 
  a.rm = b.rm
GROUP BY
  a.rm, b.obstype
ORDER BY
  a.rm
;

# add the third table, calculate length downstream
SELECT 
  a.rm, (c.rmdown-a.rm) AS dry_length, b.datet 
FROM (rivermile a
    INNER JOIN 
    observation b ON a.rm = b.rm) 
    INNER JOIN dryness c 
      ON b.id = c.id
ORDER BY
  a.rm
;

    INNER JOIN dryness c

# add the fourth table, calculate discharge
SELECT 
  a.rm, (c.rmdown-a.rm) AS dry_length, d.q as q_meas, b.datet 
FROM (rivermile a
    INNER JOIN 
    observation b ON a.rm = b.rm) 
    LEFT JOIN 
      dryness c ON b.id = c.id
    LEFT JOIN
      discharge d ON b.id = d.id
ORDER BY
  a.rm
;
