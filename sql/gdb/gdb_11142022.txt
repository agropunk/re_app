-- rm_tenths
SELECT
	rivermile.rm, 
	ST_X(rivermile.latlong) AS latitude,
	ST_Y(rivermile.latlong) AS longitude
FROM
	rivermile
WHERE round(rivermile.rm,1) = rivermile.rm;

-- tbl_dryness
SELECT 
	dat as date,
	ROUND(rm_up,1) as rm_up, 
	ROUND(rm_down,1) as rm_down
FROM 
	dry_length;
	
-- tbl_landmarks
SELECT
	feature, 
	round(rm,1) as rm
FROM
	feature
